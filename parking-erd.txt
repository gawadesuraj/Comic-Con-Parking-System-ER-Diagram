
// VEHICLE CATEGORIES
vehicle_categories [icon: truck, color: blue] {
  category_id int pk
  category_name string
}

// VEHICLES
vehicles [icon: car, color: blue] {
  vehicle_id int pk
  vehicle_number string
  category_id int
}

// PARKING ZONES

parking_zones [icon: map, color: purple] {
  zone_id int pk
  zone_name string
  level string
}

// SPOT CATEGORIES (VIP / STAFF / EV)

spot_categories [icon: tag, color: purple] {
  spot_category_id int pk
  category_name string
}

// PARKING SPOTS

parking_spots [icon: grid, color: purple] {
  spot_id int pk
  zone_id int
  spot_category_id int
  spot_number string
  is_available boolean
}

// PARKING TICKETS

tickets [icon: ticket, color: green] {
  ticket_id int pk
  ticket_number string
  issued_at timestamp
}

// PARKING SESSIONS

parking_sessions [icon: clock, color: green] {
  session_id int pk
  vehicle_id int
  spot_id int
  ticket_id int
  entry_time timestamp
  exit_time timestamp
  fee decimal
  status string
}

// PAYMENTS

payments [icon: credit-card, color: orange] {
  payment_id int pk
  session_id int
  amount decimal
  payment_method string
  payment_status string
  paid_at timestamp
}


// RELATIONSHIPS

vehicle_categories.category_id > vehicles.category_id

parking_zones.zone_id > parking_spots.zone_id
spot_categories.spot_category_id > parking_spots.spot_category_id

vehicles.vehicle_id > parking_sessions.vehicle_id
parking_spots.spot_id > parking_sessions.spot_id
tickets.ticket_id > parking_sessions.ticket_id

parking_sessions.session_id > payments.session_id
